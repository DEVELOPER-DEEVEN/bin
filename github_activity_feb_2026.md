## Appendix: Method

GraphQL query used:

```graphql
query($login:String!, $from:DateTime!, $to:DateTime!) {
  user(login:$login) {
    contributionsCollection(from:$from, to:$to) {
      pullRequestContributions(first: 100) {
        nodes { pullRequest { title url createdAt mergedAt state repository { nameWithOwner } } }
      }
      issueContributions(first: 100) {
        nodes { issue { title url createdAt state repository { nameWithOwner } } }
      }
      pullRequestReviewContributions(first: 100) {
        nodes { pullRequestReview { url createdAt state pullRequest { title url repository { nameWithOwner } } } }
      }
      commitContributionsByRepository {
        repository { nameWithOwner }
        contributions(first: 100) { nodes { occurredAt commitCount } }
      }
    }
  }
}
```
